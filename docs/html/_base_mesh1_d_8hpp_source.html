192 virtual void write_mesh_txt(const std::string &folder_name) const
193 {
194 if (!this->is_initialized)
195 throw std::runtime_error("Trying to write to .txt a mesh wich is not initialized. Call initialize_mesh() first. [" +
196 std::string(__FILE__) + ":" + std::to_string(__LINE__) + "]");
197
198 std::filesystem::create_directories("output/" + folder_name);
199
200 std::string filename;
201 if constexpr (Nature == MeshNature::SPACE)
202 filename = "output/" + folder_name + "/space_mesh.txt";
203 else if constexpr (Nature == MeshNature::VELOCITY)
204 filename = "output/" + folder_name + "/velocity_mesh.txt";
205 else
206 // Fallback for any other nature, if applicable
207 static_assert(Nature == MeshNature::SPACE || Nature == MeshNature::VELOCITY, "Unsupported MeshNature type");
208
209 std::ofstream txt_file(filename);
210 if (!txt_file.is_open())
211 {
212 std::cerr << "Failed to open file for writing: " << filename << std::endl;
213 return;
214 }
215
216 txt_file << "Computational Points (index - x_comp):\n";
217 for (std::size_t i = 0; i < x_comp.size(); ++i)
218 txt_file << i << " " << this->x_comp[i] << "\n";
219 }
